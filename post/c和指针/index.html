void F_Pointer（void）
{
    int a = 0;
   
    /** 声明指针*/
    int *p;              /* 指针变量(自动存储)未初始化的，默认值是随机的*/
    
    /** 指针赋值*/
    pa = &a;         /* int a的首地址赋值给指针变量p*/
    
    /** 解引用指针*/
    *p = 1；         /*指针变量指向的类型决定访问内存地址的延续长度*/
   
    /** 指针运算*/
    p+2;               /*指针与整数相加：从当前地址向后移动2个int单元*/
    p-2;                /*指针与整数相减：从当前地址向前移动2个int单元*/
    p++;              /*递增指针：从当前地址向后移动1个int单元*/
    p--;                /*递减指针：从当前地址向前移动1个int单元*/  
}

数组与指针

  • 数组

    数组中元素的数据类型相同，且存储地址是依次相连的。

  • 数组名

    数组名是一个地址常量，是数组的首元素的首地址。

  • 数组元素的访问方式

    数组有两种访问方式，数组下标方式只是编译器提供给编程者一种壳（语法糖）而已，实质上在编译器内部都是用指针方式来访问数组元素的。

int F_Main(void)
{
	int arr[c_arr_len] = {1,2,3,4,5};
	int sum = 0;
	sum = F_Arr(arr,c_arr_len);  /*传值调用*/
	return sum;
}

/** 函数定义：数组表示法*/
void F_Arr(int array[], int len) 
{
	int sum = 0;
	for (int i = 0; i < len; i++)
	{
		sum = sum + arr[i];
	}
	return sum;
}

/** 函数定义：指针表示法*/
void F_Array_P(int *arr, int len)   
{
	int sum = 0;
	for (int i = 0;i < len;i++)
	{
		sum = sum + *(arr + i);
	}
	return sum;  
}

结构体与指针

  • 结构体

    一组类型相同或不同的变量。

  • 结构体赋值

    声明结构体变量同时可以整体初始化；

    声明结构体变量之后，只能单独给结构成员赋值；

    两个结构体类型相同的变量可以相互赋值。

  • 结构体成员的访问方式

    结构体变量.结构体成员；

    “结构体指针->结构体成员；

    (*结构体指针).结构体成员；

/** 声明结构体类型*/
typedef struct S_Time_TYPE
{
    int  Year;                        /** 年*/
    int  Month;                    /** 月*/
    int  Day;                         /** 日*/
} S_Time;

/** 函数的返回值可以是结构体类型的值*/
S_Time F_Main(void)
{
    /** 定义结构体类型的变量（定义时初始化）*/
    S_Time S_LCU_Time = {0,0,0};
   
    /** 定义结构体类型的变量后，单独赋值*/
    S_Time S_DMI_Time;
    DMI_Time.Year = 0;
    DMI_Time.Month = 0;
    DMI_Time.Day = 0;
    
    /** 传结构体的值调用函数*/
    F_LCU_Time(S_LCU_Time);
    
    /** 传结构的地址值调用函数*/
    F_DMI_Time(&S_DMI_Time);
    
    /** 相同类型的结构体可以互相赋值*/
    S_DMI_Time = S_LCU_Time;
   
    /** 函数可以返回结构体类型的值*/
    return S_DMI_Time;
}

/** 定义带结构体形参的函数*/    
void F_LCU_Time(S_Time S_LCU_Time)
{
    S_LCU_Time.Year = 2021;
    S_LCU_Time.Month = 1;
    S_LCU_Time.Day = 11;
}

/** 定义带结构体形参的函数*/
void F_DMI_Time(S_Time *S_DMI_Time)
{
    (*S_DMI_Time).Year = 2021  /** 等效 S_DMI_Time->Year = 2021*/
    (*S_DMI_Time).Month = 1;   /** 等效 S_DMI_Time->Month = 11*/
    (*S_DMI_Time).Day = 11;     /** 等效 S_DMI_Time->Day = 1*/
}
